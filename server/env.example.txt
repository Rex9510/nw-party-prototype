# 复制为 .env 后在启动 shell 中 source / 或自行 setx（Windows）注入环境变量。
# 微信小程序 URL Scheme（wxa/generatescheme）须使用「被拉起小程序」的 AppID + AppSecret。
# 第三方「i龙岗」等若无法取得密钥，则无法通过本接口代生成，仅可继续用 H5 内 #小程序:// 短链等方式。

WECHAT_MINI_APPID=你的小程序AppId
WECHAT_MINI_SECRET=你的小程序AppSecret

# 可选：打开后的页面路径与参数（不填则微信按首页规则处理，以微信文档为准）
WECHAT_SCHEME_PATH=
WECHAT_SCHEME_QUERY=

# 当请求参数 target=ilonggang 时优先使用下面两项（便于与通用路径区分）
WECHAT_ILG_SCHEME_PATH=
WECHAT_ILG_SCHEME_QUERY=

# 可选：scheme 有效天数 1～30，默认 30
WECHAT_SCHEME_EXPIRE_DAYS=30

# 可选：服务端口，默认 8080
# PORT=8080
